59    fn execute_code(&mut self) {
60        self.is_executing = true;
61        let code = self.code.clone(); // Clone to avoid borrowing conflict
62        let result = match self.language.as_str() {
63            "BASIC" => self.execute_basic(&code),
64            "Time Warp" => self.execute_time_warp(&code),
65            "Pascal" => self.execute_pascal(&code),
66            "Prolog" => self.execute_prolog(&code),
67            "Logo" => self.execute_logo(&code),
68            "PILOT" => self.execute_pilot(&code),
69            _ => format!("Language '{}' not yet supported for execution", self.language),
70        };
71        if self.is_executing {  // Only show result if not stopped
72            self.output = format!("[Output for {}]\n{}", self.language, result);
73            // Note: No longer auto-switching to output tab since tabs are always visible
74        }
75        self.is_executing = false;
76    }

78    fn execute_basic(&mut self, code: &str) -> String {
79        let mut output = Vec::new();
80        let mut lines = Vec::new();
81
82        // Parse line-numbered BASIC program
83        for line in code.lines() {
84            let line = line.trim();
85            if line.is_empty() || line.starts_with("REM") {
86                continue;
87            }
88            if let Some(space_pos) = line.find(' ') {
89                if let Ok(line_num) = line[..space_pos].parse::<u32>() {
90                    let command = line[space_pos..].trim();
91                    lines.push((line_num, command.to_string()));
92                }
93            }
94        }
95
96        // Sort by line number
97        lines.sort_by_key(|(num, _)| *num);
98
99        for (_, command) in lines {
100            if command.to_uppercase().starts_with("PRINT") {
101                if let Some(quote_start) = command.find('"') {
102                    if let Some(quote_end) = command[quote_start + 1..].find('"') {
103                        let text = &command[quote_start + 1..quote_start + 1 + quote_end];
104                        output.push(text.to_string());
105                    }
106                } else if let Some(var_start) = command.to_uppercase().find("PRINT ") {
107                    let var_name = command[var_start + 6..].trim();
108                    if let Some(value) = self.variables.get(var_name) {
109                        output.push(value.clone());
110                    } else {
111                        output.push(format!("Undefined variable: {}", var_name));
112                    }
113                }
114            } else if command.to_uppercase().starts_with("LET") {
115                if let Some(eq_pos) = command.find('=') {
116                    let var_part = &command[4..eq_pos].trim();
117                    let value_part = command[eq_pos + 1..].trim();
118                    if let Some(var_name) = var_part.split_whitespace().last() {
119                        self.variables.insert(var_name.to_string(), value_part.to_string());
120                        output.push(format!("{} = {}", var_name, value_part));
121                    }
122                }
123            }
124        }
125
126        output.join("\n")
127    }

245    fn check_basic_syntax(&self, code: &str) -> String {
246        let mut errors = Vec::new();
247
248        for (line_num, line) in code.lines().enumerate() {
249            let line = line.trim();
250            if line.is_empty() {
251                continue;
252            }
253
254            // Check for line numbers in BASIC
255            if !line.chars().next().unwrap_or(' ').is_ascii_digit() && !line.to_uppercase().starts_with("REM") {
256                errors.push(format!("Line {}: BASIC programs should start with line numbers", line_num + 1));
257            }
258
259            // Check PRINT statements
260            if line.to_uppercase().contains("PRINT") && !line.contains("\"") && !line.chars().all(|c| c.is_alphanumeric() || c.is_whitespace() || c == '=' || c == '"') {
261                errors.push(format!("Line {}: PRINT statement syntax error", line_num + 1));
262            }
263        }
264
265        if errors.is_empty() {
266            "Syntax check passed!".to_string()
267        } else {
268            format!("Syntax errors found:\n{}", errors.join("\n"))
269        }
270    }

318    fn execute_turtle_command(&mut self, command: &str) {
319        let cmd = command.to_uppercase();
320        if cmd.starts_with("FORWARD") || cmd.starts_with("FD") {
321            if let Some(distance_str) = cmd.split_whitespace().nth(1) {
322                if let Ok(distance) = distance_str.parse::<f32>() {
323                    let rad_angle = self.turtle_state.angle.to_radians();
324                    let new_x = self.turtle_state.x + distance * rad_angle.cos();
325                    let new_y = self.turtle_state.y + distance * rad_angle.sin();
326
327                    if self.turtle_state.pen_down {
328                        self.turtle_commands.push(format!("line {} {} {} {}", self.turtle_state.x, self.turtle_state.y, new_x, new_y));
329                    }
330
331                    self.turtle_state.x = new_x;
332                    self.turtle_state.y = new_y;
333
334                    self.output.push_str(&format!("Turtle: forward {}\n", distance));
335                }
336            }
337        } else if cmd.starts_with("RIGHT") || cmd.starts_with("RT") {
338            if let Some(angle_str) = cmd.split_whitespace().nth(1) {
339                if let Ok(angle) = angle_str.parse::<f32>() {
340                    self.turtle_state.angle += angle;
341                    self.output.push_str(&format!("Turtle: right {}\n", angle));
342                }
343            }
344        } else if cmd.starts_with("LEFT") || cmd.starts_with("LT") {
345            if let Some(angle_str) = cmd.split_whitespace().nth(1) {
346                if let Ok(angle) = angle_str.parse::<f32>() {
347                    self.turtle_state.angle -= angle;
348                    self.output.push_str(&format!("Turtle: left {}\n", angle));
349                }
350            }
351        } else if cmd == "PENUP" || cmd == "PU" {
352            self.turtle_state.pen_down = false;
353            self.output.push_str("Turtle: pen up\n");
354        } else if cmd == "PENDOWN" || cmd == "PD" {
355            self.turtle_state.pen_down = true;
356            self.output.push_str("Turtle: pen down\n");
357        } else if cmd.starts_with("HOME") {
358            self.turtle_state.x = 200.0;
359            self.turtle_state.y = 200.0;
360            self.turtle_state.angle = 0.0;
361            self.output.push_str("Turtle: home\n");
362        } else if cmd.starts_with("CLEARSCREEN") || cmd == "CS" {
363            self.turtle_commands.clear();
364            self.turtle_state.x = 200.0;
365            self.turtle_state.y = 200.0;
366            self.turtle_state.angle = 0.0;
367            self.output.push_str("Turtle: clear screen\n");
368        }
369    }
